# Map

Map component.

## Basic usage

<Playground>
  <View m={4} height={400}>
    <Map apiKey={GOOGLE_MAPS_API_KEY}>
      <View {...markers[0]} mt={-25}>
        <Icon name="map-marker-alt" color="pink" size={25} />
      </View>
    </Map>
  </View>
</Playground>

## Cluster

<Playground>
  <View m={4} height={400}>
    <Map
      cluster
      apiKey={GOOGLE_MAPS_API_KEY}
      isHighlight={({id}) => id == markers[0].id}
      onChange={(bounds, listings) => console.log('onChange', {bounds, listings})}
    >
      {markers.map((props) => (
        <Map.Marker
          {...props}
          key={props.id}
          onSelect={() => console.log('onSelect', props.id)}
        >
          {String(props.text)}
        </Map.Marker>
      ))}
    </Map>
  </View>
</Playground>

## Multi marker

<Playground>
  <View m={4} height={400}>
    <Map
      cluster
      apiKey={GOOGLE_MAPS_API_KEY}
      isHighlight={({id}) => id === 1}
    >
      {['A', 'B', 'C'].map((name, i) => (
        <Map.Marker {...markers[0]} key={name} id={i}>
          {'Marker ' + name}
        </Map.Marker>
      ))}
    </Map>
  </View>
</Playground>
